    LOCAL PROCEDURE CalcPercentage@3(PartAmount@1000 : Decimal;Base@1001 : Decimal) : Decimal;
    BEGIN
      IF Base <> 0 THEN
        EXIT(100 * PartAmount / Base);

      EXIT(0);
    END;

    PROCEDURE MakeNegativeZero@4(Amount@1000 : Decimal) : Decimal;
    BEGIN
      IF Amount < 0 THEN
        EXIT(0);
      EXIT(Amount);
    END;

    BEGIN
    END.
