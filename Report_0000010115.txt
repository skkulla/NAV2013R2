    PROCEDURE ProcessInvoices@1(PaymentEntry@1000 : Record 25);
    BEGIN
      { search for invoices paid off by this payment }
      EntryAppMgt.GetAppliedVendEntries(TempAppliedEntry,PaymentEntry,TRUE);
      WITH TempAppliedEntry DO BEGIN
        { search for invoices with 1099 amounts }
        SETFILTER("Document Type",'%1|%2',"Document Type"::Invoice,"Document Type"::"Credit Memo");
        SETFILTER("IRS 1099 Amount",'<>0');
        CASE FormType OF
          1:
            SETRANGE("IRS 1099 Code",'MISC-','MISC-99');
          2:
            SETRANGE("IRS 1099 Code",'DIV-','DIV-99');
          3:
            SETRANGE("IRS 1099 Code",'INT-','INT-99');
        END;
        IF FIND('-') THEN
          REPEAT
            Calculate1099Amount(TempAppliedEntry,PaymentEntry,"Amount to Apply");
          UNTIL NEXT = 0;
      END;
    END;

    PROCEDURE Calculate1099Amount@2(InvoiceEntry@1000 : Record 25;PaymentEntry@1001 : Record 25;AppliedAmount@1002 : Decimal);
    BEGIN
      InvoiceEntry.CALCFIELDS(Amount);
      Invoice1099Amount := -AppliedAmount * InvoiceEntry."IRS 1099 Amount" / InvoiceEntry.Amount;
      MagMediaManagement.UpdateLines(InvoiceEntry,FormType,EndLine,InvoiceEntry."IRS 1099 Code",Invoice1099Amount);
    END;

    LOCAL PROCEDURE ValidVendor@1020001(VAR FilteredVendor@1020002 : Record 23;VendorNo@1020003 : Code[20]) : Boolean;
    VAR
      CheckVendor@1020000 : Record 23;
    BEGIN
      IF NOT FilteredVendor.HASFILTER THEN
        EXIT(TRUE);
      CheckVendor.COPYFILTERS(FilteredVendor);
      CheckVendor.SETRANGE("No.",VendorNo);
      EXIT(NOT CheckVendor.ISEMPTY);
    END;

    BEGIN
    END.
