Public Function IsNewPage(HeaderNoCopyNo as Object) As Boolean
newPage = FALSE
Dim group1 as Object
Dim group2 as Object
group1=Cstr(Choose(1,Split(Cstr(HeaderNoCopyNo),Chr(177))))
group2=Cstr(Choose(2,Split(Cstr(HeaderNoCopyNo),Chr(177))))
If Not (group1 = currentgroup1)
    newPage = TRUE
    currentgroup1 = group1
    currentgroup2 = group2
    ELSE 
      If Not (group2 = currentgroup2)
        newPage = TRUE 
        currentgroup2 = group2
     End If
  End If
  Return newPage
End Function
