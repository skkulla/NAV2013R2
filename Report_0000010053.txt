    LOCAL PROCEDURE GetCurrencyRecord@1020000(VAR Currency@1020000 : Record 4;CurrencyCode@1020001 : Code[10]);
    BEGIN
      IF CurrencyCode = '' THEN BEGIN
        CLEAR(Currency);
        Currency.Description := GLSetup."LCY Code";
        Currency."Amount Rounding Precision" := GLSetup."Amount Rounding Precision";
      END ELSE
        IF Currency.Code <> CurrencyCode THEN
          Currency.GET(CurrencyCode);
    END;

    LOCAL PROCEDURE GetCurrencyCaptionCode@1020001(CurrencyCode@1020000 : Code[10]) : Text[80];
    BEGIN
      IF PrintAmountsInLocal THEN BEGIN
        IF CurrencyCode = '' THEN
          EXIT('101,1,' + Text002);

        GetCurrencyRecord(Currency,CurrencyCode);
        EXIT('101,4,' + STRSUBSTNO(Text002,Currency.Description));;
      END;
        EXIT('');
    END;

    BEGIN
    END.
