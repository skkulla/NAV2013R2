    PROCEDURE QFormat@1(Qty@1000 : Decimal) : Text[250];

      TempStr := FORMAT(ROUND(Qty));
      j := STRPOS(TempStr,'.');
      IF j = 0 THEN
        TempStr := TempStr + '.00'
